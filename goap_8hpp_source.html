310 T blackboard;
311 float cost;

313 std::optional<size_t> action_taken_idx;

315 };
316
317 using node_ptr = std::shared_ptr<node_type>;
318
319 struct node_compare {
320 bool operator()(const node_ptr& a, const node_ptr& b) const {
321 return a->cost > b->cost;
322 }
323 };
324
325 std::priority_queue<node_ptr, std::vector<node_ptr>, node_compare> open_set;
326 std::unordered_set<T> closed_set;
327
328 open_set.push(std::make_shared<node_type>(node_type{
329 .blackboard = initital_blackboard,
330 .cost = 0.0f,
331 .action_taken_idx = std::nullopt,
332 .parent = nullptr
333 }));
334
335 for (
336 size_t iteration = 0;
337 !open_set.empty() && (max_iterations == 0 || iteration < max_iterations);
338 ++iteration
339 ) {
340 auto current_node = open_set.top();
341 open_set.pop();
342
343 if (current_node->blackboard == goal_blackboard) {
344 auto p = plan<T>();
345 p.m_actions = std::move(actions);
346
347 while (current_node->parent != nullptr) {
348 auto action_idx = current_node->action_taken_idx.value();
349 p.m_plan.push(action_idx);
350 current_node = current_node->parent;
351 }
352
353 return p;
354 }
355
356 if (!closed_set.contains(current_node->blackboard)) {
357 closed_set.insert(current_node->blackboard);
358
359 for (size_t action_idx = 0; action_idx < actions.size(); action_idx++) {
360 auto& action = actions[action_idx];
361
362 if (action->check_preconditions(current_node->blackboard)) {
363 auto next_blackboard = current_node->blackboard;
364 action->apply_effects(next_blackboard, true);
365 auto next_cost = current_node->cost + action->cost(current_node->blackboard);
366
367 if (!closed_set.contains(next_blackboard)) {
368 open_set.push(std::make_shared<node_type>(node_type{
369 .blackboard = next_blackboard,
